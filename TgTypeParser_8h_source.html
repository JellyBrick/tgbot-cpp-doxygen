282  std::shared_ptr<T> tryParseJson(JsonToTgTypeFunc<T> parseFunc, const boost::property_tree::ptree& data, const std::string& keyName) const {
283  auto treeItem = data.find(keyName);

327  std::vector<std::vector<std::shared_ptr<T>>> result;

342  template<typename T>
343  std::string parseArray(TgTypeToJsonFunc<T> parseFunc, const std::vector<std::shared_ptr<T>>& objects) const {
344  if (objects.empty())
345  return "";
346  std::string result;
347  result += '[';
348  for (const std::shared_ptr<T>& item : objects) {
349  result += (this->*parseFunc)(item);
350  result += ',';
351  }
352  result.erase(result.length() - 1);
353  result += ']';
354  return result;
355  }

372  template<typename T>
373  std::string parse2DArray(TgTypeToJsonFunc<T> parseFunc, const std::vector<std::vector<std::shared_ptr<T>>>& objects) const {
374  if (objects.empty())
375  return "";
376  std::string result;
377  result += '[';
