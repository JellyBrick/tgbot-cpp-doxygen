63 ## Examples
64 
65 Simple echo bot which sends everything it receives:
66 ```cpp
67 #include <stdio.h>
68 #include <tgbot/tgbot.h>
69 
70 int main() {
71  TgBot::Bot bot("PLACE YOUR TOKEN HERE");
72  bot.getEvents().onCommand("start", [&bot](TgBot::Message::Ptr message) {
73  bot.getApi().sendMessage(message->chat->id, "Hi!");
74  });
75  bot.getEvents().onAnyMessage([&bot](TgBot::Message::Ptr message) {
76  printf("User wrote %s\n", message->text.c_str());
77  if (StringTools::startsWith(message->text, "/start")) {
78  return;
79  }
80  bot.getApi().sendMessage(message->chat->id, "Your message is: " + message->text);
81  });
82  try {
83  printf("Bot username: %s\n", bot.getApi().getMe()->username.c_str());
84  TgBot::TgLongPoll longPoll(bot);
85  while (true) {
86  printf("Long poll started\n");
87  longPoll.start();
88  }
89  } catch (TgBot::TgException& e) {
90  printf("error: %s\n", e.what());
91  }
92  return 0;
93 }
94 ```
